## 字典序的第K小数字


给定整数 n 和 k，返回  [1, n] 中字典序第 k 小的数字。

```
impl Solution {
    /// 寻找 1 到 n 中字典序的第 k 小的数字
    ///
    /// 字典序（lexicographical order）也称为"词典顺序"
    /// 例如：1, 10, 100, 101, ..., 11, 12, ..., 2, 20, ...
    ///
    /// # 参数
    /// - `n`: 最大数字（包含）
    /// - `k`: 要查找的序号（1-indexed）
    ///
    /// # 返回
    /// 第 k 小的数字（按字典序）
    ///
    /// # 示例
    /// ```
    /// assert_eq!(Solution::find_kth_number(13, 2), 10);
    /// // 字典序: 1, 10, 11, 12, 13, 2, 3, ..., 9
    /// //          ↑ 第 2 个是 10
    /// ```
    ///
    /// # 算法核心
    /// 将数字看作 10 叉树的节点，按先序遍历（深度优先）顺序，
    /// 使用"跳过子树"的技巧快速定位第 k 个节点。
    ///
    /// 十叉树结构示例（n=100）：
    /// ```
    ///                   1                    2    3 ... 9
    ///                 / | \
    ///               10 11 12 ...
    ///               /
    ///             100
    /// ```
    pub fn find_kth_number(n: i32, mut k: i32) -> i32 {
        // 闭包：统计以 node 为根的子树包含多少个节点（不超过 n）
        //
        // 子树节点规律（以 node=1, n=100 为例）：
        // 第 0 层：1
        // 第 1 层：10-19（10 个节点）
        // 第 2 层：100-199（实际只有 100 在范围内）
        // 总计：1 + 10 + 1 = 12 个节点
        let count_subtree_size = |node: i32| -> i32 {
            let n = n as i64;  // 转换为 i64 避免计算时溢出
            let mut size = 0;
            let mut left = node as i64;   // 当前层的最小值
            let mut right = node as i64 + 1; // 当前层的最大值 + 1（用于计算数量）

            // 逐层统计
            while left <= n {
                // 当前层的节点数量 = min(right, n+1) - left
                // 使用 n+1 是因为我们想要包含 n 本身
                size += (right.min(n + 1) - left) as i32;

                // 移动到下一层（每个数字后面加 0）
                left *= 10;
                right *= 10;
            }

            size
        };

        let mut node = 1;  // 从根节点 1 开始
        k -= 1;            // 减去当前访问的 node 节点

        // 继续寻找，直到找到第 k 个节点
        while k > 0 {
            let size = count_subtree_size(node);

            if size <= k {
                // 情况 1: 当前子树的所有节点数 <= 剩余要跳过的数量
                // 说明第 k 个节点不在当前子树中
                node += 1;      // 移动到当前节点的右侧兄弟节点
                k -= size;      // 减去跳过整个子树的大小
            } else {
                // 情况 2: 第 k 个节点在当前子树中
                node *= 10;     // 移动到当前节点的第一个子节点（即数字末尾加 0）
                k -= 1;         // 减去当前节点本身
            }
        }

        node
    }
}
```
