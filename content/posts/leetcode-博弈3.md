## 预测赢家

给你一个整数数组 nums 。玩家 1 和玩家 2 基于这个数组设计了一个游戏。

玩家 1 和玩家 2 轮流进行自己的回合，玩家 1 先手。开始时，两个玩家的初始分值都是 0 。每一回合，玩家从数组的任意一端取一个数字（即，nums[0] 或 nums[nums.length - 1]），取到的数字将会从数组中移除（数组长度减 1 ）。玩家选中的数字将会加到他的得分上。当数组中没有剩余数字可取时，游戏结束。

如果玩家 1 能成为赢家，返回 true 。如果两个玩家得分相等，同样认为玩家 1 是游戏的赢家，也返回 true 。你可以假设每个玩家的玩法都会使他的分数最大化。


```
impl Solution {
    pub fn predict_the_winner(nums: Vec<i32>) -> bool {
        // ==================== 问题理解 ====================
        // 这是一个两人轮流从数组两端取数的游戏
        // 每个玩家都希望自己最终的总分比对方高
        // 两个玩家都采取最优策略
        // 返回先手玩家是否能赢（分数 >= 对手分数）

        // ==================== 动态规划定义 ====================
        // dp[i][j] 表示在子数组 nums[i..=j] 中，当前玩家与对手的最大分差
        // 正值表示当前玩家能比对手多得多少分
        // 负值表示当前玩家会比对手少得多少分

        let n = nums.len();
        // 创建 n x n 的 DP 表，初始化为 0
        let mut dp = vec![vec![0; n]; n];

        // ==================== 初始化对角线 ====================
        // 当 i == j 时，子数组只有一个数字 nums[i]
        // 当前玩家拿走这个数字，对手得 0 分
        // 所以当前玩家领先 nums[i] 分
        for i in 0..n {
            dp[i][i] = nums[i];
        }

        // ==================== 状态转移 ====================
        // 按区间长度从小到大计算
        // i 从 n-2 递减到 0，确保计算 dp[i][j] 时，dp[i+1][j] 和 dp[i][j-1] 已经计算过
        for i in (0..n-1).rev() {  // 从倒数第二行开始，向上遍历
            for j in i+1..n {      // j 从 i+1 到 n-1，向右遍历
                // 当前玩家有两种选择：
                // 1. 取左边的数 nums[i]
                //    获得 nums[i] 分，然后对手在子数组 nums[i+1..=j] 中成为当前玩家
                //    对手会获得 dp[i+1][j] 的分差优势（相对于他当时的对手，也就是现在的我）
                //    所以相对于当前玩家，在对手回合我会损失 dp[i+1][j] 分
                //    最终分差 = nums[i] - dp[i+1][j]
                //
                // 2. 取右边的数 nums[j]
                //    获得 nums[j] 分，然后对手在子数组 nums[i..=j-1] 中成为当前玩家
                //    对手会获得 dp[i][j-1] 的分差优势
                //    最终分差 = nums[j] - dp[i][j-1]
                //
                // 当前玩家会选择对自己最有利（分差最大）的策略
                dp[i][j] = (nums[i] - dp[i+1][j]).max(nums[j] - dp[i][j-1]);
            }
        }

        // ==================== 返回结果 ====================
        // dp[0][n-1] 表示在整个数组 nums[0..=n-1] 中，先手玩家能获得的分差
        // 如果分差 >= 0，说明先手玩家分数 >= 对手分数，先手获胜
        dp[0][n-1] >= 0
    }
}
```
