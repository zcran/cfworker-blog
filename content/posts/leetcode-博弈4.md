## 猫和老鼠

两位玩家分别扮演猫和老鼠，在一张 无向 图上进行游戏，两人轮流行动。

图的形式是：graph[a] 是一个列表，由满足 ab 是图中的一条边的所有节点 b 组成。

老鼠从节点 1 开始，第一个出发；猫从节点 2 开始，第二个出发。在节点 0 处有一个洞。

在每个玩家的行动中，他们 必须 沿着图中与所在当前位置连通的一条边移动。例如，如果老鼠在节点 1 ，那么它必须移动到 graph[1] 中的任一节点。

此外，猫无法移动到洞中（节点 0）。

然后，游戏在出现以下三种情形之一时结束：

如果猫和老鼠出现在同一个节点，猫获胜。
如果老鼠到达洞中，老鼠获胜。
如果某一位置重复出现（即，玩家的位置和移动顺序都与上一次行动相同），游戏平局。
给你一张图 graph ，并假设两位玩家都都以最佳状态参与游戏：

如果老鼠获胜，则返回 1；
如果猫获胜，则返回 2；
如果平局，则返回 0 。


```
impl Solution {
    pub fn cat_mouse_game(g: Vec<Vec<i32>>) -> i32 {
        let n = g.len();
        if n == 0 { return 0; }
        if n <= 2 { return 1; } // 只有0,1节点时老鼠必胜

        // 状态总数: n个老鼠位置 * n个猫位置 * 2个回合(0老鼠走,1猫走)
        let sz = n * n * 2;
        let mut d = vec![0u8; sz]; // 胜负结果: 0未知,1老鼠胜,2猫胜
        let mut deg = vec![0u16; sz]; // 每个状态的出度(能转移到的状态数)
        let mut q = Vec::with_capacity(sz); // BFS队列

        // ===== 1. 初始化每个状态的出度 =====
        for m in 0..n {
            for c in 0..n {
                let i = m * n * 2 + c * 2; // 老鼠回合的索引
                deg[i] = g[m].len() as u16; // 老鼠回合: 老鼠可以走的邻居数
                // 猫回合: 猫可以走的邻居数(不能走到0号洞)
                deg[i + 1] = g[c].iter().filter(|&&x| x != 0).count() as u16;
            }
        }

        // ===== 2. 标记已知终局状态 =====
        for m in 0..n {
            for c in 0..n {
                let i = m * n * 2 + c * 2;
                // 老鼠在洞里(m==0) 或 猫鼠同位置(m==c) 都是终局
                if m == 0 || m == c {
                    let r = if m == 0 { 1 } else { 2 }; // 洞:老鼠胜, 同位置:猫胜
                    d[i] = r;     // 老鼠回合的结果
                    d[i + 1] = r; // 猫回合的结果
                    // 加入队列用于逆向传播
                    q.push((m, c, 0, r));
                    q.push((m, c, 1, r));
                }
            }
        }

        // ===== 3. 逆向BFS传播胜负 =====
        let mut f = 0;
        while f < q.len() {
            let (m, c, t, r) = q[f]; f += 1;
            // 如果已经推出初始状态的结果,直接返回
            if m == 1 && c == 2 && t == 0 { return r as i32; }

            // 前驱状态的回合: 当前t=0则前驱是猫走(t=1), 当前t=1则前驱是老鼠走(t=0)
            let (pt, pti) = (t ^ 1, (t ^ 1) as usize);

            if pt == 0 { // 前驱是老鼠回合
                // 遍历前驱老鼠可能的位置pm(能一步走到当前m)
                for &pm in &g[m] {
                    let pm = pm as usize;
                    let i = pm * n * 2 + c * 2 + pti;
                    if d[i] == 0 { // 还未确定结果
                        if r == 1 {
                            // 当前老鼠胜,前驱老鼠可以一步走到胜态 → 前驱老鼠胜
                            d[i] = 1;
                            q.push((pm, c, pt, 1));
                        } else {
                            // 当前猫胜,前驱老鼠的这一步走向败态,减少出度
                            deg[i] -= 1;
                            if deg[i] == 0 {
                                // 所有走法都导向猫胜 → 前驱老鼠败(猫胜)
                                d[i] = 2;
                                q.push((pm, c, pt, 2));
                            }
                        }
                    }
                }
            } else { // 前驱是猫回合(pt==1)
                // 遍历前驱猫可能的位置pc(能一步走到当前c)
                for &pc in &g[c] {
                    let pc = pc as usize;
                    if pc == 0 { continue; } // 猫不能从洞出发
                    let i = m * n * 2 + pc * 2 + pti;
                    if d[i] == 0 {
                        if r == 2 {
                            // 当前猫胜,前驱猫可以一步走到胜态 → 前驱猫胜
                            d[i] = 2;
                            q.push((m, pc, pt, 2));
                        } else {
                            // 当前老鼠胜,前驱猫的这一步走向败态,减少出度
                            deg[i] -= 1;
                            if deg[i] == 0 {
                                // 所有走法都导向老鼠胜 → 前驱猫败(老鼠胜)
                                d[i] = 1;
                                q.push((m, pc, pt, 1));
                            }
                        }
                    }
                }
            }
        }

        // 返回初始状态(老鼠1,猫2,老鼠先走)的结果
        d[1 * n * 2 + 2 * 2] as i32
    }
}
```
